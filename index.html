### CSS (styles.css) @import url('https://fonts.googleapis.com/css2?family=Montserrat:wght@100;200;300;400;500;600;700;800;900&display=swap'); .gradient-container { background-image: linear-gradient(120deg, #3498db 0%, #9b59b6 100%); padding: 20px; border-radius: 10px; box-shadow: 0 0 10px rgba(0,0,0,0.3); } .header { font-weight: bold; font-size: 24px; color: white; } .repository-list { max-width: 600px; margin: auto; padding: 20px; border: 1px solid #ddd; border-radius: 10px; box-shadow: 0 0 10px rgba(0,0,0,0.3); } .repository-card { background-color: white; padding: 10px; border-radius: 10px; box-shadow: 0 0 5px rgba(0,0,0,0.2); margin-bottom: 10px; } .repository-card h2 { font-weight: bold; font-size: 18px; margin-top: 0; } .repository-card p { font-size: 14px; } .repository-card a { text-decoration: none; color: #337ab7; } ### JavaScript (script.js) // Get the search input and button elements const searchInput = document.getElementById('search-input'); const searchButton = document.getElementById('search-button'); const repositoryList = document.getElementById('repository-list'); // Add an event listener to the search button searchButton.addEventListener('click', () => { const searchString = searchInput.value.trim(); if (searchString) { fetch(`https://api.github.com/search/repositories?q=${searchString}`) .then(response => response.json()) .then(data => { const repositories = data.items; const listHTML = repositories .map(repo => `

${repo.name}

${repo.description} (${repo.owner.login})

`) .join(''); repositoryList.innerHTML = listHTML; }) .catch(error => console.error('Error:', error)); } }); This code sets up a GitHub Tools Hub with a search bar, a button to submit the search, and a list to display the search results. The CSS uses a dark/glassmorphism style, and the JavaScript uses the Fetch API to retrieve data from the GitHub API. The results are displayed as cards in the list.
