⭐️ ${repo.stargazers_count}

${repo.keywords.includes('APK') ? `` : ''}
`; return html; } // On page load, fetch the top 10 Termux and Python repositories getTermuxRepositories() .then(termuxRepos => { getPythonRepositories() .then(pythonRepos => { // Get the top 10 repositories from both Termux and Python const topRepos = [...termuxRepos.items.slice(0, 5), ...pythonRepos.items.slice(0, 5)]; topRepos.forEach(repo => { const html = renderRepoCard(repo); repoList.innerHTML += html; }); }) .catch(error => console.error(error)); }) .catch(error => console.error(error)); // Search button click event handler searchButton.addEventListener('click', async (e) => { e.preventDefault(); const searchTerm = searchInput.value.trim(); if (searchTerm) { const results = await getRepositories(searchTerm); repoList.innerHTML = ''; // Clear the repository list results.items.forEach(repo => { const html = renderRepoCard(repo); repoList.innerHTML += html; }); } }); Please note that this JavaScript code assumes you have the following HTML elements in your page:
